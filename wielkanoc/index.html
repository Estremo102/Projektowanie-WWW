function easter(IY) {
  let A = IY % 19;
  let B = Math.floor(IY / 100);
  let C = IY % 100;
  let H = (19 * A + B - Math.floor(B / 4) - Math.floor((B + 1 - Math.floor((B + 8) / 25)) / 3) + 15) % 30;
  let L = (32 + 2 * (B % 4 + Math.floor(C / 4)) - H - (C % 4)) % 7;
  let M = Math.floor((A + 11 * H + 22 * L) / 451);
  let N = H + L - 7 * M + 114;
  let month = Math.floor(N / 31); // 3 = marzec, 4 = kwiecień
  let day = 1 + (N % 31);
  return { month, day };
}
